K = 10000
meanhat = np.zeros((K,19))
for dset in range(K):
    for idx,year in enumerate(range(2000,2019,1)):
        yr1 = '01-01-'+str(year)
        yr2 = '12-01-'+str(year)
        querystr = "Date>='"+yr1+"' & Date<='"+yr2+"'"
        meanhat[dset,idx] = kwh.query(querystr).sample(n=12,replace=True)['Average kWh'].mean()
